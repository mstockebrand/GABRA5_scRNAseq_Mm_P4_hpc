#  CLUSTERING
##  Clustering the cells
#### find neighbors
    Scn2a.P4 <- FindNeighbors(Scn2a.P4, reduction = "integrated.dr", dims = 1:last_pc_id)
#### find clusters
    Scn2a.P4 <- FindClusters(Scn2a.P4, resolution = 0.8) ## default resolution = 0.8
##  UMAP non-linear dimensionality reduction
    Scn2a.P4 <- RunUMAP(Scn2a.P4, dims = 1:last_pc_id, reduction = "integrated.dr")
